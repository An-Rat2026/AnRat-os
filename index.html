@echo off
setlocal enabledelayedexpansion
chcp 1251 > nul
title AnRat MAX Premium
color 0A

:: ========================================
:: ПРОВЕРКА ЛИЦЕНЗИИ (два типа ключей)
:: ========================================
set "LICENSE_FILE=%userprofile%\AnRat_License.key"
set "IS_PREMIUM=0"
set "LICENSE_TYPE="
set "HOURS_LEFT=0"
set "MINUTES_LEFT=0"

if exist "%LICENSE_FILE%" (
    for /f "tokens=1,2 delims=|" %%a in (%LICENSE_FILE%) do (
        set "LICENSE_TYPE=%%a"
        set "EXPIRE_TIME=%%b"
    )
    if "!LICENSE_TYPE!"=="FOREVER" set "IS_PREMIUM=1"
    if "!LICENSE_TYPE!"=="TRIAL" (
        for /f "tokens=1-3 delims=.:" %%a in ("%time%") do (
            set /a "CURRENT_SECONDS=%%a*3600+%%b*60+%%c"
        )
        if !CURRENT_SECONDS! LSS !EXPIRE_TIME! (
            set "IS_PREMIUM=1"
            set /a "HOURS_LEFT=(EXPIRE_TIME-CURRENT_SECONDS)/3600"
            set /a "MINUTES_LEFT=((EXPIRE_TIME-CURRENT_SECONDS)%%3600)/60"
        ) else del "%LICENSE_FILE%" 2>nul
    )
)

:: ========================================
:: АВТОМАТИЧЕСКОЕ ОБНОВЛЕНИЕ
:: ========================================
set "VERSION=2.1"
set "UPDATE_URL=https://raw.githubusercontent.com/An-Rat2026/AnRat-os/main/version.txt"
set "DOWNLOAD_URL=https://raw.githubusercontent.com/An-Rat2026/AnRat-os/main/AnRat_MAX_Premium.bat"

set "LAST_CHECK_FILE=%temp%\AnRat_last_check.txt"
set "CURRENT_DATE=%date%"

if not exist "%LAST_CHECK_FILE%" (
    echo %CURRENT_DATE% > "%LAST_CHECK_FILE%"
    goto :check_update
)

set /p LAST_CHECK=<"%LAST_CHECK_FILE%"
if not "%LAST_CHECK%"=="%CURRENT_DATE%" (
    echo %CURRENT_DATE% > "%LAST_CHECK_FILE%"
    goto :check_update
)
goto :skip_update

:check_update
echo.
echo   🔄 Проверка обновлений...
ping -n 2 127.0.0.1 > nul
powershell -Command "try { $response = Invoke-WebRequest -Uri '%UPDATE_URL%' -UseBasicParsing -TimeoutSec 5; $response.Content } catch { '0' }" > "%temp%\version_check.txt" 2>nul
set /p SERVER_VERSION=<"%temp%\version_check.txt"
if "%SERVER_VERSION%"=="0" goto :skip_update
if "%SERVER_VERSION%"=="%VERSION%" goto :skip_update
cls
echo.
echo ========================================
echo        🔄 ДОСТУПНО ОБНОВЛЕНИЕ!
echo ========================================
echo   Текущая версия: %VERSION%
echo   Новая версия: %SERVER_VERSION%
echo.
echo   [1] ✅ ОБНОВИТЬ СЕЙЧАС
echo   [2] ⏰ НАПОМНИТЬ ПОЗЖЕ
echo.
set /p update_choice="Выберите: "
if "%update_choice%"=="1" goto :do_update
goto :skip_update

:do_update
cls
echo.
echo ========================================
echo        🔄 УСТАНОВКА ОБНОВЛЕНИЯ
echo ========================================
echo.
echo Скачиваю новую версию...
powershell -Command "Invoke-WebRequest -Uri '%DOWNLOAD_URL%' -OutFile '%temp%\AnRat_new.bat'" 2>nul
copy /y "%temp%\AnRat_new.bat" "%~f0" > nul
echo.
echo ========================================
echo        ✅ ОБНОВЛЕНИЕ УСТАНОВЛЕНО!
echo ========================================
echo.
echo   Перезапустите программу.
pause
exit

:skip_update

reg add "HKCU\Software\Microsoft\Windows\CurrentVersion\ContentDeliveryManager" /v "SubscribedContent-338387Enabled" /t REG_DWORD /d 0 /f >nul

reg add "HKCU\Software\Microsoft\Windows\CurrentVersion\ContentDeliveryManager" /v "SubscribedContent-353698Enabled" /t REG_DWORD /d 0 /f >nul

reg add "HKLM\SOFTWARE\Policies\Microsoft\Windows\CloudContent" /v "DisableWindowsConsumerFeatures" /t REG_DWORD /d 1 /f >nul

:isolated_desktop
if "%IS_PREMIUM%"=="0" (
    echo.
    echo ========================================
    echo        ❌ ДОСТУП ЗАПРЕЩЁН
    echo ========================================
    echo.
    echo   🔐 Изолированный рабочий стол доступен только в Premium версии.
    call :back_to_menu
    goto main_menu
)
cls
echo.
echo ========================================
echo        🔐 ИЗОЛИРОВАННЫЙ РАБОЧИЙ СТОЛ
echo ========================================
echo.
echo   ВНИМАНИЕ! Будет создан временный пользователь.
echo   После выхода из сессии он будет удалён.
echo.
set /p "isolate_choice=Запустить? (y/n): "
if /i not "%isolate_choice%"=="y" goto main_menu
set "TEMP_USER=AnRatSecure"
set /p "user_pass=Придумайте пароль: "
net user %TEMP_USER% "%user_pass%" /add /expires:never >nul 2>&1
net localgroup Guests %TEMP_USER% /add >nul 2>&1
reg add "HKLM\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Winlogon\SpecialAccounts\UserList" /v %TEMP_USER% /t REG_DWORD /d 0 /f >nul 2>&1
echo.
echo ✅ Изолированная среда создана. Переключение...
timeout /t 2 /nobreak >nul
tsdiscon.exe
runas /user:%TEMP_USER% "explorer.exe"
net user %TEMP_USER% /delete >nul 2>&1
reg delete "HKLM\SOFTWARE\Microsoft\Windows NT\CurrentVersion\Winlogon\SpecialAccounts\UserList" /v %TEMP_USER% /f >nul 2>&1
echo ✅ Временный пользователь удалён.
call :back_to_menu
goto main_menu

echo ========================================
echo        🔓 СОХРАНЁННЫЕ ПАРОЛИ WI-FI
echo ========================================
echo.
for /f "tokens=2 delims=:" %%a in ('netsh wlan show profiles ^| findstr ":" ^| findstr /v "профиль"') do (
    set "ssid=%%a"
    set "ssid=!ssid:~1!"
    for /f "tokens=2 delims=:" %%b in ('netsh wlan show profile name="!ssid!" key=clear ^| findstr "Содержимое ключа"') do (
        echo 📡 !ssid! — пароль: %%b
    )
)
call :back_to_menu
goto main_menu

echo ========================================
echo        🌡️ МОНИТОРИНГ ТЕМПЕРАТУРЫ
echo ========================================
echo.
for /f "tokens=2 delims==" %%a in ('wmic /namespace:\\root\wmi path MSAcpi_ThermalZoneTemperature get CurrentTemperature /value 2^>nul') do (
    set /a "temp=%%a / 10 - 273"
    echo 🔥 Температура процессора: !temp!°C
)
if "%temp%"=="" echo ⚠️ Датчики температуры не найдены.
call :back_to_menu

powercfg -setactive 8c5e7fda-e8bf-4a96-9a85-a6e23a8c635c >nul
reg add "HKCU\Software\Microsoft\Windows\CurrentVersion\GameDVR" /v "AppCaptureEnabled" /t REG_DWORD /d 0 /f >nul
reg add "HKCU\Software\Microsoft\Windows NT\CurrentVersion\Multimedia\SystemProfile" /v "SystemResponsiveness" /t REG_DWORD /d 0 /f >nul
sc config "SysMain" start= disabled >nul
sc config "WSearch" start= disabled >nul
sc config "wuauserv" start= disabled >nul
taskkill /f /im OneDrive.exe >nul 2>&1
taskkill /f /im Spotify.exe >nul 2>&1
taskkill /f /im Skype.exe >nul 2>&1

sc config "WSearch" start= auto >nul

echo.
echo ========================================
echo        🔑 АКТИВАЦИЯ ЛИЦЕНЗИИ

echo   💎 Вечный ключ: AnRat_Max

set /p "license_key=Введите ключ: "
if "%license_key%"=="AnRat_Max" (
    echo FOREVER|0 > "%LICENSE_FILE%"
    set "IS_PREMIUM=1"
    set "LICENSE_TYPE=FOREVER"
    echo ✅ Вечная лицензия активирована!
    call :back_to_menu
    goto main_menu
)
if "%license_key%"=="AnRat_Max_Free" (
    if exist "%LICENSE_FILE%" (
        for /f "tokens=1,2 delims=|" %%a in (%LICENSE_FILE%) do set "EXISTING_TYPE=%%a"
        if "!EXISTING_TYPE!"=="TRIAL" (echo ❌ Пробный период уже использован. & call :back_to_menu & goto main_menu)
        if "!EXISTING_TYPE!"=="FOREVER" (echo ✅ У вас уже есть вечная лицензия. & call :back_to_menu & goto main_menu)
    )
    for /f "tokens=1-3 delims=.:" %%a in ("%time%") do (
        set /a "EXPIRE_SECONDS=%%a*3600+%%b*60+%%c+86400"
    )
    echo TRIAL|!EXPIRE_SECONDS! > "%LICENSE_FILE%"
    set "IS_PREMIUM=1"
    set "LICENSE_TYPE=TRIAL"
    set "HOURS_LEFT=24"
    set "MINUTES_LEFT=0"
    echo ✅ Пробная лицензия на 24 часа!
    call :back_to_menu
    goto main_menu

:password_manager
if "%IS_PREMIUM%"=="0" (echo ❌ Только Premium & call :back_to_menu & goto main_menu)
cls
echo 1. Сохранить пароль
echo 2. Показать все
echo 3. Найти
echo 4. Удалить
echo 5. Генератор
set /p act="Выбор: "
if "%act%"=="1" (
    set /p site="Сайт: "
    set /p log="Логин: "
    set /p pass="Пароль: "
    mkdir "%userprofile%\AnRat_Passwords" 2>nul
    (echo Сайт: %site% & echo Логин: %log% & echo Пароль: %pass% & echo Дата: %date%) > "%userprofile%\AnRat_Passwords\%site%.txt"
    echo Сохранено.
)
if "%act%"=="2" dir "%userprofile%\AnRat_Passwords\*.txt"
if "%act%"=="3" (
    set /p search="Поиск: "
    type "%userprofile%\AnRat_Passwords\%search%.txt" 2>nul || echo Не найдено
)
if "%act%"=="4" (
    set /p delsite="Удалить: "
    del "%userprofile%\AnRat_Passwords\%delsite%.txt" 2>nul && echo Удалено
)
if "%act%"=="5" (
    set "passgen="
    for /l %%i in (1,1,12) do set "passgen=!passgen!!random!"
    echo Сгенерированный пароль: !passgen!
    echo !passgen! | clip
)
call :back_to_menu
goto main_menu

cls

if "%t%"=="4" reg add "HKCU\Software\Microsoft\Windows\DWM" /v "AccentColor" /t REG_DWORD /d 0x00aa00 /f

goto main_menu

:lock_folder
set /p fold="Путь к папке: "
if not exist "%fold%" (echo Папки нет & call :back_to_menu & goto main_menu)
set /p pass="Пароль: "
set /p conf="Повтор: "
if not "%pass%"=="%conf%" (echo Не совпадает & call :back_to_menu & goto main_menu)
echo %pass% > "%fold%\AnRat_Lock.key"
attrib +h "%fold%\AnRat_Lock.key" & attrib +h "%fold%"
echo Папка скрыта и защищена.
call :back_to_menu
goto main_menu

:unlock_folder
set /p fold="Путь к папке: "
if not exist "%fold%\AnRat_Lock.key" (echo Не защищена & call :back_to_menu & goto main_menu)
set /p pass="Пароль: "
set /p saved=<"%fold%\AnRat_Lock.key"
if not "%pass%"=="%saved%" (echo Неверно & call :back_to_menu & goto main_menu)
attrib -h "%fold%" & attrib -h "%fold%\AnRat_Lock.key"
echo Папка открыта.
call :back_to_menu
goto main_menu

:show_locked
echo Защищённые папки:
for /f "tokens=*" %%d in ('dir /s /b C:\AnRat_Lock.key 2^>nul') do echo %%~dpd
call :back_to_menu
goto main_menu

:change_password
set /p fold="Путь к папке: "
if not exist "%fold%\AnRat_Lock.key" (echo Не защищена & call :back_to_menu & goto main_menu)
set /p oldp="Старый пароль: "
set /p savedp=<"%fold%\AnRat_Lock.key"
if not "%oldp%"=="%savedp%" (echo Неверно & call :back_to_menu & goto main_menu)
set /p newp="Новый пароль: "
set /p confp="Повтор: "
if not "%newp%"=="%confp%" (echo Не совпадает & call :back_to_menu & goto main_menu)
echo %newp% > "%fold%\AnRat_Lock.key"
echo Пароль изменён.
call :back_to_menu
goto main_menu

:end
exit

  3. Вставь код (Ctrl+V)
  4. Нажми Файл → Сохранить как
  5. Тип файла: выбери «Все файлы (*.*)»
  6. Имя файла: напиши AnRat_MAX_Premium.bat
  7. Кодировка: выбери ANSI (не UTF-8!)
  8. Нажми Сохранить
  9. Запусти файл правой кнопкой → От имени администратора

✅ После запуска следуй инструкциям в меню.
